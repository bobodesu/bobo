/* windows 系统 */
clip < ~/.ssh/id_rsa.pub

/* macOS 系统 */
pbcopy < ~/.ssh/id_rsa.pub

/* Linux 系统 */
sudo apt-get install xclip
xclip -sel clip < ~/.ssh/id_rsa.pub
